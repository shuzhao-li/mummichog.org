import pandas as pd
import seaborn as sns
infile = "../results/pvals_pos.txt"
mydata = pd.read_table(infile)
sns.set()
d = mydata.loc[mydata.pval < 0.01, :].iloc[:, 5:]
g = sns.clustermap(d,col_cluster=False, z_score=0, figsize=(10, 10))
sns.plt.show()
